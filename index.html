// OpenAI API调用示例 - 通过4Z API中转服务
const axios = require('axios');

const API_KEY = 'your_4Z_api_key'; // 替换为您的4Z API密钥
const API_URL = 'https://api.zzzzapi.com/v1/chat/completions';

async function callOpenAIAPI() {
  try {
    const response = await axios.post(
      API_URL,
      {
        model: 'gpt-4', // 可选GPT-4、GPT-3.5-turbo等所有OpenAI模型
        messages: [
          { role: 'system', content: '你是一个专业的AI助手' },
          { role: 'user', content: '请解释OpenAI API的主要功能' }
        ],
        temperature: 0.7,
        max_tokens: 800
      },
      {
        headers: {
          'Authorization': `Bearer ${API_KEY}`,

  } catch (error) {
    console.error('API调用错误:', error.response?.data || error.message);
  }
}

callOpenAIAPI();
